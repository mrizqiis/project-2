CREATE TABLE pengembalian (
    id_pengembalian INT AUTO_INCREMENT PRIMARY KEY,
    id_sewa INT NOT NULL,                     -- relasi ke tabel sewa/transaksi
    tanggal_kembali DATE NOT NULL,            -- tanggal mobil dikembalikan oleh pelanggan
    tanggal_sekarang DATE NOT NULL,           -- tanggal sistem saat pencatatan
    denda DECIMAL(10,2) DEFAULT 0.00,         -- denda jika terlambat
    total_bayar DECIMAL(10,2) NOT NULL,       -- total yang dibayar
    keterangan TEXT,                          -- catatan tambahan
    dibuat_pada TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (id_sewa) REFERENCES sewa(id_sewa)
);
